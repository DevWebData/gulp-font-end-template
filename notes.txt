function cacheBust(src, dest) {
    var cbString = new Date().getTime();
    return gulp
      .src(src)
      .pipe(
        replace(/cb=\d+/g, function() {
          return "cb=" + cbString;
        })
      )
      .pipe(gulp.dest(dest));
  }
